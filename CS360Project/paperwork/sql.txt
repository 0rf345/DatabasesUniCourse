-- Individual Table
create table individual
(
	id           int(6) not null auto_increment primary key,
	accid        varchar(9) as '00-'+right('000000'+cast(id as varchar(6)), 6) persistent,
	usern        varchar(32) not null,
	userp        varchar(64) not null,
	expdate      date,
	creditlmt    decimal(15,2) not null default 0,
	currdebt     decimal(15,2) not null default 0,
	availcredit  decimal(15,2) not null default 0,
	fname        varchar(64) not null,
	lname        varchar(64) not null
) engine = InnoDB;

-- Company Table
create table company
(
	id           int(6) not null auto_increment primary key,
	accid        varchar(9) as '01-'+right('000000'+cast(id as varchar(6)), 6) persistent,
	usern        varchar(32) not null,
	userp        varchar(64) not null,
	expdate      date,
	creditlmt    decimal(15,2) not null default 0,
	currdebt     decimal(15,2) not null default 0,
	availcredit  decimal(15,2) not null default 0,
	companyname  varchar(64) not null
) engine = InnoDB;

-- Company Employee Table
create table company_employee
	id     int(3) not null auto_increment primary key,
	empid  varchar(9) as company_accid+'-'+right('000'+cast(id as varchar(3)), 3) persistent,
	fname  varchar(64) not null,
	lname  varchar(64) not null,
	constraint `fk_company_accid`
		foreign key (company_accid)
			references company(accid)
			on update cascade
) engine = InnoDB;

-- Merchant Table
create table merchant
(
	id           int(6) not null auto_increment primary key,
	accid        varchar(9) as '02-'+right('000000'+cast(id as varchar(6)), 6) presistent,
	usern        varchar(32) not null,
	userp        varchar(64) not null,
	expdate      date,
	creditlmt    decimal(15,2) not null default 0,
	currdebt     decimal(15,2) not null default 0,
	availcredit  decimal(15,2) not null default 0,
	fname        varchar(64) not null,
	lname        varchar(64) not null,
	commission   decimal(5,2) not null default 5.00,
	profit       decimal(15,2) not null
) engine = InnoDB;

-- Individual Transaction Table
create table individual_transaction
(
	id         int(8) not null auto_increment primary key,
	trsctid    varchar(11) as '00-'+right('00000000'+cast(id as varchar(8)), 8) persistent,
	trsctdate  date not null,
	amount     decimal(15,2) not null default 0,
	ischarge   boolean not null default false
	constraint `fk_indiv_accid`
		foreign key (indiv_accid)
			references individual(accid)
			on update cascade
	constraint `fk_merch_accid`
		foreign key (merch_accid)
			references merchant(accid)
			on update cascade
) engine = InnoDB;

-- Company Transaction Table
create table company_transaction
(
	id         int(8) not null auto_increment primary key,
	trsctid    varchar(11) as '01-'+right('00000000'+cast(id as varchar(8)), 8) persistent,
	trsctdate  date not null,
	amount     decimal(15,2) not null default 0,
	ischarge   boolean not null default false
	constraint `fk_company_accid`
		foreign key (company_accid)
			references company(accid)
			on update cascade
	constraint `fk_merch_accid`
		foreign key (merch_accid)
			references merchant(accid)
			on update cascade
) engine = InnoDB;
